    id: str,
    name: str,
    age: int,
    weight: float,
    height: float

impl Patient {
    fn bmi(self) -> float {
        return bmi(self.weight, self.height)
    }

    fn bsa(self) -> float {
        return bsa(self.weight, self.height * 100)  // height in cm
    }

    fn dose(self, mg_per_kg: float) -> float {
        return dose_per_kg(mg_per_kg, self.weight)
    }
}

enum Diagnosis {
    Normal,
    Underweight,
    Overweight,
    Obese
}

fn classify_bmi(patient: Patient) -> Diagnosis {
    let bmi = patient.bmi()
    match bmi {
        _ if bmi < 18.5 => Underweight,
        _ if bmi < 25.0 => Normal,
        _ if bmi < 30.0 => Overweight,
        _ => Obese
    }
}

    Underweight => print("Consider nutritional support"),
    Overweight => print("Recommend lifestyle changes"),
    Obese => print("Medical intervention needed")
}
